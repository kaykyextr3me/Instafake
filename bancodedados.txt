create database db_instafake;

use db_instafake;

create table usuario_dadospessoais(
	id integer primary key auto_increment,
    celular varchar(24),
    email varchar(128),
    nome_completo varchar(128),
    nome_usuario varchar(64),
    senha varchar (128));
    
alter table usuario_dadospessoais add foto_perfil varchar(256);
alter table usuario_dadospessoais add biografia varchar(256);

CREATE TABLE publicacoes (
    id int PRIMARY KEY auto_increment,
    name_post varchar(256),
    data_post varchar(24),
    fk_usuario_dadospessoais_id int,
    FOREIGN key (fk_usuario_dadospessoais_id ) REFERENCES usuario_dadospessoais (id)
)
